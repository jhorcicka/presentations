\begin{frame}{What do we (programmers) do?}
  \begin{block}{Read}
    \begin{itemize}
      \item Fast reading. % do not say it in your head, eyes-stetch exercises, keep going, read multiple words, lines, skimming, scanning, using a pointer
      \item Correct colors and fonts. % so that it does not make your eyes tired quickly
      \item Suitable chair. % not to damage your posture
      \item Get up regularly. % do not sit all the time, short breaks every hour
    \end{itemize}
  \end{block}
  \begin{block}{Write}
    \begin{itemize}
      \item Fast writing. % 10 vs 2 fingers, 70-80 words/min, 130 words/min
      \item Good keyboard. % my laptop vs keyboard
      \item Efficient tools and SW. % vim, of course
    \end{itemize}
  \end{block}
\end{frame}
